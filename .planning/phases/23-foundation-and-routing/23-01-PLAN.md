<objective>
Create the config CLI module (lib/config.cjs) and Inner Voice state file module (subsystems/reverie/state.cjs) -- the two foundational data access modules that all subsequent Phase 23 work depends on.

Purpose: These modules provide the data layer for the Reverie subsystem. The config module enables `dynamo config get/set` for feature flags (reverie.mode).     case 'config': {
    ```javascript
      const configModule = require(resolve('lib', 'config.cjs'));
      const subCmd = restArgs[0];
      if (subCmd === 'get') {
        const key = restArgs[1];
        if (!key) { error('Usage: dynamo config get <key>'); return; }
        const value = configModule.get(key);
        if (value === undefined) {
          process.stderr.write('(not set)\n');
        } else {
          const display = typeof value === 'object' ? JSON.stringify(value, null, 2) : String(value);
          process.stderr.write(display + '\n');
        }
      } else if (subCmd === 'set') {
        const key = restArgs[1];
        const val = restArgs[2];
        if (!key || val === undefined) { error('Usage: dynamo config set <key> <value>'); return; }
        try {
          const written = configModule.set(key, val);
          process.stderr.write(key + ' = ' + JSON.stringify(written) + '\n');
        } catch (e) {
          error(e.message);
        }
      } else {
        error('Usage: dynamo config <get|set> <key> [value]');
      }
      break;
    }
    ```
